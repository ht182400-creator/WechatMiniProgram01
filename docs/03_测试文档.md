# tests/test_indicators.py
import pytest
import pandas as pd
from factors import TechnicalFactors

class TestTechnicalFactors:
    
    @pytest.fixture
    def sample_data(self):
        """生成测试数据"""
        dates = pd.date_range('2024-01-01', periods=100)
        df = pd.DataFrame({
            'open': [100 + i * 0.1 for i in range(100)],
            'high': [105 + i * 0.1 for i in range(100)],
            'low': [95 + i * 0.1 for i in range(100)],
            'close': [100 + i * 0.12 for i in range(100)],
            'volume': [1000000] * 100
        }, index=dates)
        return df
    
    def test_moving_averages(self, sample_data):
        factors = TechnicalFactors(sample_data)
        df = factors.add_moving_averages()
        
        assert 'sma_5' in df.columns
        assert 'sma_20' in df.columns
        assert 'ema_12' in df.columns
        
        # 验证计算正确性
        assert pd.notna(df['sma_5'].iloc[-1])
        assert pd.notna(df['sma_20'].iloc[-1])
    
    def test_rsi_calculation(self, sample_data):
        factors = TechnicalFactors(sample_data)
        df = factors.add_rsi(sample_data)
        
        assert 'rsi_14' in df.columns
        rsi = df['rsi_14'].dropna()
        assert all((rsi >= 0) & (rsi <= 100))
    
    def test_macd_calculation(self, sample_data):
        factors = TechnicalFactors(sample_data)
        df = factors.add_macd(sample_data)
        
        assert 'macd' in df.columns
        assert 'macd_signal' in df.columns
        assert 'macd_hist' in df.columns

import pandas as pd
import numpy as np

def generate_mock_kline(n=250, start_price=100):
    """生成模拟K线数据"""
    dates = pd.date_range('2024-01-01', periods=n)
    
    returns = np.random.normal(0.001, 0.02, n)
    prices = start_price * np.cumprod(1 + returns)
    
    df = pd.DataFrame({
        'open': prices * (1 + np.random.uniform(-0.01, 0.01, n)),
        'high': prices * (1 + np.random.uniform(0, 0.02, n)),
        'low': prices * (1 - np.random.uniform(0, 0.02, n)),
        'close': prices,
        'volume': np.random.uniform(1e6, 1e7, n)
    }, index=dates)
    
    return df

    runs-on: ubuntu-latest

          pip install -r backend/requirements.txt

      - name: Run tests

          pytest tests/ -v --cov=backend

          cd frontend/web
          npm install
