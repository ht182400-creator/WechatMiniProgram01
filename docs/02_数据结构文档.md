## 1. 数据库设计

### 1.1 数据库选型

- **开发环境**: SQLite
- **生产环境**: PostgreSQL

### 1.2 表结构

#### 1.2.1 stocks - 股票基本信息表

```sql
CREATE TABLE stocks (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    code VARCHAR(10) NOT NULL UNIQUE,      -- 股票代码
    name VARCHAR(100) NOT NULL,             -- 股票名称
    market VARCHAR(10),                    -- 市场: sh, sz, bj
    industry VARCHAR(100),                 -- 所属行业
    list_date VARCHAR(10),                 -- 上市日期
    is_enabled BOOLEAN DEFAULT 1,         -- 是否启用
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_stocks_code ON stocks(code);
```

#### 1.2.2 daily_kline - 日线K线数据表

```sql
CREATE TABLE daily_kline (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    code VARCHAR(10) NOT NULL,             -- 股票代码
    date VARCHAR(10) NOT NULL,             -- 交易日期
    open FLOAT,                           -- 开盘价
    high FLOAT,                           -- 最高价
    low FLOAT,                            -- 最低价
    close FLOAT,                          -- 收盘价
    volume FLOAT,                         -- 成交量
    amount FLOAT,                         -- 成交额
    change FLOAT,                         -- 涨跌额
    pct_change FLOAT,                    -- 涨跌幅
    turnover FLOAT,                      -- 换手率
    adjust_flag VARCHAR(5) DEFAULT 'qfq', -- 复权类型
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(code, date)
);

CREATE INDEX idx_kline_code ON daily_kline(code);
CREATE INDEX idx_kline_date ON daily_kline(date);
```

#### 1.2.3 technical_indicators - 技术指标缓存表

```sql
CREATE TABLE technical_indicators (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    code VARCHAR(10) NOT NULL,
    date VARCHAR(10) NOT NULL,
    indicators JSON,                      -- 技术指标JSON
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(code, date)
);

CREATE INDEX idx_indicators_code ON technical_indicators(code);
```

#### 1.2.4 backtest_records - 回测记录表

```sql
CREATE TABLE backtest_records (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    strategy_name VARCHAR(50) NOT NULL,   -- 策略名称
    stock_code VARCHAR(10),               -- 股票代码
    start_date VARCHAR(10),               -- 回测开始日期
    end_date VARCHAR(10),                -- 回测结束日期
    initial_cash FLOAT,                  -- 初始资金
    final_capital FLOAT,                 -- 最终资金
    total_return FLOAT,                  -- 总收益率
    annualized_return FLOAT,             -- 年化收益率
    max_drawdown FLOAT,                  -- 最大回撤
    sharpe_ratio FLOAT,                  -- 夏普比率
    win_rate FLOAT,                      -- 胜率
    total_trades INTEGER,                -- 交易次数
    params JSON,                         -- 策略参数
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

#### 1.2.5 prediction_records - 预测记录表

```sql
CREATE TABLE prediction_records (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    code VARCHAR(10) NOT NULL,            -- 股票代码
    date VARCHAR(10) NOT NULL,            -- 预测日期
    model_name VARCHAR(50),             -- 模型名称
    current_price FLOAT,                -- 当前价格
    predicted_price FLOAT,              -- 预测价格
    predicted_direction VARCHAR(10),     -- 预测方向: up, down, neutral
    confidence FLOAT,                   -- 置信度
    actual_price FLOAT,                 -- 实际价格(后续更新)
    accuracy FLOAT,                     -- 准确率(后续计算)
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_prediction_code ON prediction_records(code);
```

#### 1.2.6 system_config - 系统配置表

```sql
CREATE TABLE system_config (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    config_key VARCHAR(100) NOT NULL UNIQUE, -- 配置键
    config_value TEXT,                       -- 配置值
    description VARCHAR(500),                -- 配置说明
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

## 2. 数据模型 (Pydantic)

### 2.1 请求模型

```python
# 回测请求
class BacktestRequest(BaseModel):
    code: str                          # 股票代码
    strategy_name: str                 # 策略名称
    start_date: Optional[str] = None   # 开始日期
    end_date: Optional[str] = None    # 结束日期
    initial_cash: float = 100000       # 初始资金
    params: Optional[str] = None       # 策略参数字符串
```

### 2.2 响应模型

```python
# 回测结果响应
class BacktestResultResponse(BaseModel):
    success: bool
    code: str
    strategy: str
    result: BacktestMetrics
    equity_curve: List[EquityPoint]

class BacktestMetrics(BaseModel):
    final_capital: float
    total_return: float
    annualized_return: float
    max_drawdown: float
    sharpe_ratio: float
    win_rate: float
    total_trades: int
```

## 5. 策略信号数据结构

```python
@dataclass
class TradingSignal:
    date: str              # 日期
    signal: SignalType     # 信号类型 BUY/SELL/HOLD
    price: float           # 信号价格
    quantity: int          # 交易数量
    reason: str            # 信号原因
    confidence: float      # 置信度 0-1

@dataclass
class Trade:
    date: str              # 交易日期
    action: str            # buy/sell
    code: str              # 股票代码
    price: float           # 成交价格
    quantity: int          # 成交数量
    commission: float      # 手续费
    slippage: float        # 滑点成本
    total_amount: float    # 总金额
```
